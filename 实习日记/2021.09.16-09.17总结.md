##### js命名风格：驼峰和下划线互相转换

```

let newKey=key.replace(/_([a-z])/g, (p, m) => m.toUpperCase())
//驼峰转下划线
let newKey = key.replace(/([A-Z])/g, (p, m) => `_${m.toLowerCase()}`)
```

deconstruct(obj: any) {
        let array = [];
        if (obj) {
            for (const prop in obj) {
                let objValue = obj[prop];
                array.push({
                        name:prop

                }
            }

    }
